<CodeBlock language="sql" title="Active Users by Department">
  SELECT d.name, COUNT(u.sys_id) 
  FROM department d
  JOIN sys_user u ON d.sys_id = u.department_id
  WHERE u.active = true
  GROUP BY d.name
</CodeBlock>
